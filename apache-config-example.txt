<VirtualHost *:80>
    ProxyPreserveHost On
    ServerName backups.turbinux.com
    ServerAlias www.backups.turbinux.com
    
    # Proxy all requests to Node.js application
    ProxyPass / http://SERVER_IP:5000/
    ProxyPassReverse / http://SERVER_IP:5000/
    
    # Enable required Apache modules:
    # a2enmod proxy
    # a2enmod proxy_http
    
    # Additional security headers
    Header always set X-Frame-Options "SAMEORIGIN"
    Header always set X-XSS-Protection "1; mode=block"
    Header always set X-Content-Type-Options "nosniff"
    
    # Logging configuration
    ErrorLog ${APACHE_LOG_DIR}/backups-error.log
    CustomLog ${APACHE_LOG_DIR}/backups-access.log combined
</VirtualHost>
